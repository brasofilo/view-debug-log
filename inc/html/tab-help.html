    define('WP_DEBUG', true);
    if ( WP_DEBUG ) {
        define( 'WP_DEBUG_LOG', false );
        define('WP_DEBUG_DISPLAY', false);
        $path = realpath( $_SERVER["DOCUMENT_ROOT"] . '/..' ) . '/debug.log';
        @ini_set( 'log_errors', 'On' );
        @ini_set( 'error_log', $path );
        error_reporting(0);
        @ini_set('display_errors', 0);
    }
